00064 ConversionResult ConvertUTF32toUTF16 (
00065         const UTF32** sourceStart, const UTF32* sourceEnd, 
00066         UTF16** targetStart, UTF16* targetEnd, ConversionFlags flags) {
00067     ConversionResult result = conversionOK;
00068     const UTF32* source = *sourceStart;
00069     UTF16* target = *targetStart;
00070     while (source < sourceEnd) {
00071         UTF32 ch;
00072         if (target >= targetEnd) {
00073             result = targetExhausted; break;
00074         }
00075         ch = *source++;
00076         if (ch <= UNI_MAX_BMP) { /* Target is a character <= 0xFFFF */
00077             /* UTF-16 surrogate values are illegal in UTF-32; 0xffff or 0xfffe are both reserved values */
00078             if (ch >= UNI_SUR_HIGH_START && ch <= UNI_SUR_LOW_END) {
00079                 if (flags == strictConversion) {
00080                     --source; /* return to the illegal value itself */
00081                     result = sourceIllegal;
00082                     break;
00083                 } else {
00084                     *target++ = UNI_REPLACEMENT_CHAR;
00085                 }
00086             } else {
00087                 *target++ = (UTF16)ch; /* normal case */
00088             }
00089         } else if (ch > UNI_MAX_LEGAL_UTF32) {
00090             if (flags == strictConversion) {
00091                 result = sourceIllegal;
00092             } else {
00093                 *target++ = UNI_REPLACEMENT_CHAR;
00094             }
00095         } else {
00096             /* target is a character in range 0xFFFF - 0x10FFFF. */
00097             if (target + 1 >= targetEnd) {
00098                 --source; /* Back up source pointer! */
00099                 result = targetExhausted; break;
00100             }
00101             ch -= halfBase;
00102             *target++ = (UTF16)((ch >> halfShift) + UNI_SUR_HIGH_START);
00103             *target++ = (UTF16)((ch & halfMask) + UNI_SUR_LOW_START);
00104         }
00105     }
00106     *sourceStart = source;
00107     *targetStart = target;
00108     return result;
00109 }

00113 ConversionResult ConvertUTF16toUTF32 (
00114         const UTF16** sourceStart, const UTF16* sourceEnd, 
00115         UTF32** targetStart, UTF32* targetEnd, ConversionFlags flags) {
00116     ConversionResult result = conversionOK;
00117     const UTF16* source = *sourceStart;
00118     UTF32* target = *targetStart;
00119     UTF32 ch, ch2;
00120     while (source < sourceEnd) {
00121         const UTF16* oldSource = source; /*  In case we have to back up because of target overflow. */
00122         ch = *source++;
00123         /* If we have a surrogate pair, convert to UTF32 first. */
00124         if (ch >= UNI_SUR_HIGH_START && ch <= UNI_SUR_HIGH_END) {
00125             /* If the 16 bits following the high surrogate are in the source buffer... */
00126             if (source < sourceEnd) {
00127                 ch2 = *source;
00128                 /* If it's a low surrogate, convert to UTF32. */
00129                 if (ch2 >= UNI_SUR_LOW_START && ch2 <= UNI_SUR_LOW_END) {
00130                     ch = ((ch - UNI_SUR_HIGH_START) << halfShift)
00131                         + (ch2 - UNI_SUR_LOW_START) + halfBase;
00132                     ++source;
00133                 } else if (flags == strictConversion) { /* it's an unpaired high surrogate */
00134                     --source; /* return to the illegal value itself */
00135                     result = sourceIllegal;
00136                     break;
00137                 }
00138             } else { /* We don't have the 16 bits following the high surrogate. */
00139                 --source; /* return to the high surrogate */
00140                 result = sourceExhausted;
00141                 break;
00142             }
00143         } else if (flags == strictConversion) {
00144             /* UTF-16 surrogate values are illegal in UTF-32 */
00145             if (ch >= UNI_SUR_LOW_START && ch <= UNI_SUR_LOW_END) {
00146                 --source; /* return to the illegal value itself */
00147                 result = sourceIllegal;
00148                 break;
00149             }
00150         }
00151         if (target >= targetEnd) {
00152             source = oldSource; /* Back up source pointer! */
00153             result = targetExhausted; break;
00154         }
00155         *target++ = ch;
00156     }
00157     *sourceStart = source;
00158     *targetStart = target;
00159 #ifdef CVTUTF_DEBUG

00161     fprintf(stderr, "ConvertUTF16toUTF32 illegal seq 0x%04x,%04x\n", ch, ch2);
00162     fflush(stderr);
00163 }
00164 #endif
00165     return result;

00304 /*
00305  * Utility routine to tell whether a sequence of bytes is legal UTF-8.
00306  * This must be called with the length pre-determined by the first byte.
00307  * If not calling this from ConvertUTF8to*, then the length can be set by:
00308  *  length = trailingBytesForUTF8[*source]+1;
00309  * and the sequence is illegal right away if there aren't that many bytes
00310  * available.
00311  * If presented with a length > 4, this returns false.  The Unicode
00312  * definition of UTF-8 goes up to 4-byte sequences.
00313  */
00314 
00315 /*exodus*/
00316 /*inline*/

00318 static Boolean isLegalUTF8(const UTF8 *source, int length) {
00319     UTF8 a;
00320     const UTF8 *srcptr = source+length;
00321     switch (length) {
00322     default: return false;
00323         /* Everything else falls through when "true"... */
00324     case 4: if ((a = (*--srcptr)) < 0x80 || a > 0xBF) return false;
00325     case 3: if ((a = (*--srcptr)) < 0x80 || a > 0xBF) return false;
00326     case 2: if ((a = (*--srcptr)) > 0xBF) return false;
00327 
00328         switch (*source) {
00329             /* no fall-through in this inner switch */
00330             case 0xE0: if (a < 0xA0) return false; break;
00331             case 0xED: if (a > 0x9F) return false; break;
00332             case 0xF0: if (a < 0x90) return false; break;
00333             case 0xF4: if (a > 0x8F) return false; break;
00334             default:   if (a < 0x80) return false;
00335         }
00336 
00337     case 1: if (*source >= 0x80 && *source < 0xC2) return false;
00338     }
00339     if (*source > 0xF4) return false;
00340     return true;
00341 }
