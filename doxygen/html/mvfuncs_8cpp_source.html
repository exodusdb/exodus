00192 {
00193         /*pick round rounds positive .5 up to 1 and negative .5 down to -1 etc
00194         -1.0=-1
00195         -0.9=-1
00196         -0.5=-1
00197         -0.1= 0
00198          0  = 0
00199          0.1= 0
00200          0.5= 1.0
00201          0.9= 1.0
00202          1.0= 1.0
00203         */

00205         //if n=0 could save the integer conversion here but would require mentality to save BOTH int and double
00206         //currently its possible since space is reserved for both but this may change
00207 
00208         THISIS(L"var var::round() const")
00209         THISISNUMERIC()
00210 
00211         double result;
00212         if (var_mvtype&pimpl::MVTYPE_INT)
00213         {
00214                 if (not ndecimals)
00215                         return var_mvint;
00216                 result=var_mvint;
00217         }
00218         else
00219                 result=var_mvdbl;
00220 
00221         //scale it up (or down)
00222         double scale=pow(10.0,ndecimals);
00223         result*=scale;
00224 
00225         //round .5 up and -.5 down to nearest integer as per pick financial rounding concept
00226         long long int result2;
00227         if (result>=0)
00228                 result2=(long long int)(result+0.5);
00229         else
00230                 result2=(long long int)(result-0.5);
00231 
00232         //scale it back down (or back up)
00233         result=result2/scale;

00476 var& var::trimmer(const wchar_t* trimchar)
00477 {
00478 
00479         //reimplement with boost string trim_if algorithm
00480         //http://www.boost.org/doc/libs/1_39_0/doc/html/string_algo/reference.html
00481 
00482         THISIS(L"var& var::trimmer(const wchar_t* trimchar)")
00483         THISISSTRING()
00484 
00485         //find the first non blank
00486         std::wstring::size_type start_pos;
00487         start_pos=var_mvstr.find_first_not_of(trimchar);
00488 
00489         //if all blanks return empty string
00490         if (start_pos==std::wstring::npos)
00491         {
00492                 *this=L"";
00493                 return *this;
00494         }
00495 
00496         //erase leading spaces
00497         if (start_pos!=0) var_mvstr.erase(0,start_pos);
00498 
00499         //find the last non blank
00500         std::wstring::size_type end_pos;
00501         //end_pos=var_mvstr.find_last_not_of(trimchar);
00502         end_pos=var_mvstr.find_last_not_of(trimchar,var_mvstr.length()-1);
00503 
00504         //erase trailing spaces
00505         var_mvstr.erase(end_pos+1);
00506 
00507         //find the starting position of any embedded spaces
00508         start_pos=std::wstring::npos;
00509         while (true)
00510         {
00511                 //find a space
00512                 start_pos=var_mvstr.find_last_of(trimchar,start_pos);
00513 
00514                 //if no (more) spaces then return the string
00515                 if (start_pos==std::wstring::npos||start_pos<=0) return *this;
00516 
00517                 //find the first non-space thereafter
00518                 end_pos=var_mvstr.find_last_not_of(trimchar,start_pos-1);
00519 
00520                 //if first non space character is not one before the space
00521                 if (end_pos<start_pos-1)
00522                 {
00523                         var_mvstr.erase(end_pos+1,start_pos-end_pos-1);
00524                 }
00525                 if (end_pos<=0) break;
00526                 start_pos=end_pos-1;
00527         }
00528         return *this;
00529 
00530 }
